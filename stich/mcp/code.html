      "args": [

        "@techblog/mcp-server"

      "env": {
        "TECHBLOG_API_KEY": "sk-pro-••••••••7f2b",
        "TECHBLOG_ENDPOINT": "https://api.techblog.admin/v1"
      },
      "version": 1.2
